## Solution

Language: **java**  /  Runtime: 0 ms

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode convertBiNode(TreeNode root) {
        inOrder(root);
        return resultNode;
    }

    TreeNode resultNode;

    void inOrder(TreeNode node){
        /**
         * 用flag旗标来记录第一个左子树为空的点,即搜索树最左下角的点,并把它作为头结点
         */
        if(node!=null){
            inOrder(node.left);
            visit(node);
            inOrder(node.right);
        }
    }

    TreeNode pre = null;
    void visit(TreeNode cur){
        if(pre == null){
            resultNode = cur;
        }else {
            cur.left = null;
            pre.right = cur;
        }
        pre=cur;
    }
}
```
